CREATE TABLE tb_amostra (
       id_amostra INT8 GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
        data_de_entrada DATE,
        nome_amostra VARCHAR(255) NOT NULL,
        nota_fiscal VARCHAR(255) NOT NULL,
        status_amostra VARCHAR(255) NOT NULL,
        tipo_amostra VARCHAR(255) NOT NULL,
        validade DATE NOT NULL,
        id_solicitacao_de_analise int8 NOT NULL
    );
